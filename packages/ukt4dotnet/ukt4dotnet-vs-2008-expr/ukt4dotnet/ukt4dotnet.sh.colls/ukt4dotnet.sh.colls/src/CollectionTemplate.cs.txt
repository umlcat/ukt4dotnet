    /// <summary>
    /// A template for entity classes.
    /// </summary>
    public class <classname> : CollectionClass
    {
        #region "fields"
            // ...

            // ...
        #endregion "fields"

        #region "properties"
            // ...

            // ...
        #endregion "properties"

        #region "support"
            private bool F_<classname>_IsReady;

            public <classname>()
            {
                // --> clear status
                F_<classname>_IsReady = false;
				// --> clear fields
				// ...
            } // <classname>()

            public override bool IsReady()
            {
                bool Result = F_<classname>_IsReady;
                return Result;
            } // bool IsReady(...)
        #endregion "support"

        #region "constructors"
            // ...

            public override Int64 Create()
            {
                Int64 Result = 0;

                Result = base.Create();

                // --> update status
                F_<classname>_IsReady = true;

                return Result;
            } // public override Int64 Create(...)
            
            public override Int64 Destroy()
            {
                Int64 Result = 0;

                // --> update status
                F_<classname>_IsReady = false;

                Result = base.Destroy();

                return Result;
            } // public override Int64 Destroy(...)

            // ...
        #endregion "constructors"

        #region "methods"
			// ...

            public override bool IsEmpty()
            {
                bool Result = false;

                // ...

                return Result;
            } // void Clear()

            public override void Clear()
            {
                // ...
            } // void Clear()

			// ...
        #endregion "methods"

        // ...

    } // class <classname>
